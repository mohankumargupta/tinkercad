You can get away with not using Keyboard shortcuts in Tinkercad, but some are just invaluable to make your workflow in Tinkercad faster.


## Editor Shortcuts

<ShortcutMenu>
    <h5 class="text-[#559ef6]">ESSENTIAL</h5>
    <Shortcut label="Undo">
      <Kbd filled>Ctrl</Kbd>
      <Kbd>Z</Kbd>
    </Shortcut>
    <Shortcut label="Duplicate">
      <Kbd filled>Ctrl</Kbd>
      <Kbd>D</Kbd>
    </Shortcut>
    <Shortcut label="Drop to Workplane">
      <Kbd>D</Kbd>
    </Shortcut>
    <Shortcut label="Uniform resizing while clicking shape white square corner">
      <Kbd filled>Shift</Kbd>
      <Kbd>◻</Kbd>
    </Shortcut>
    <Shortcut label="Uniform resizing from center while clicking shape white square corner">
      <Kbd filled>Alt</Kbd>
      <Kbd filled>Shift</Kbd>
      <Kbd>◻</Kbd>
    </Shortcut>
    <Shortcut label="Nudge object right(by grid size)">
      <Kbd>→</Kbd>
    </Shortcut>
    <Shortcut label="Nudge object right by 10 times the grid size">
      <Kbd filled>Shift</Kbd>
      <Kbd>→</Kbd>
    </Shortcut>
    <Shortcut label="Nudge object off workplace vertically">
      <Kbd filled>Ctrl</Kbd>
      <Kbd>↑</Kbd>
    </Shortcut>
    <Shortcut label="When resizing or rotating, put focus on dimensions textbox">
      <Kbd filled>TAB</Kbd>
    </Shortcut>

    <h5 class="text-[#559ef6]">NICE TO KNOW</h5>
    <Shortcut label="Make Transparent">
      <Kbd>T</Kbd>
    </Shortcut>
    <Shortcut label="Panning Shift+RightClick+Move workplane">
      <Kbd filled>Shift</Kbd>
      <Kbd>RC</Kbd>
    </Shortcut>


</ShortcutMenu>
